class Solution {
public:
    // Returns total hours needed if Koko eats
    // 'speed' bananas per hour.
    long long totalHours(vector<int>& piles, int speed) {
        long long hours = 0;
        for(int pile : piles) {
            // Ceiling of pile / speed
            hours += (pile + speed - 1) / speed;
        }
        return hours;
    }
    int minEatingSpeed(vector<int>& piles, int h) {
        int low = 1;
        int high = *max_element(piles.begin(), piles.end());
        while(low <= high) {
            int mid = low + (high - low) / 2;
            if(totalHours(piles, mid) <= h) {
                // Current speed works.
                // Try a smaller speed.
                high = mid - 1;
            }
            else {
                // Too slow.
                // Increase speed.
                low = mid + 1;
            }
        }
        // Smallest valid speed.
        return low;
    }
};
