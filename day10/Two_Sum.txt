class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        map<int, int> mpp;
        int n = nums.size();
        for(int i = 0; i < n; i++) {
            int a = nums[i];
            int needed = target - a;
            if(mpp.find(needed) != mpp.end()) {
                return {mpp[needed], i};
            }
            mpp[a] = i;
        }
        return {};
    }
};
