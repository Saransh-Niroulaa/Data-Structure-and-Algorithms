class Solution {
  public:
    int floorSqrt(int n) {
        int low =0; 
        int high= n;
        int ans ;
        while(low <=high){
            int mid= (low+high)/2;
            if (mid*mid==n){return mid;}
            else if(mid*mid>n){
                high =mid-1;
                ans=high;
            }
            else {
                low=mid+1;
            }
        }
        return ans;
    }
};
