class Solution {
public:

    int maxSubArray(vector<int>& nums) {
        int sum = 0, max_i = nums[0];
        for (int i = 0; i<nums.size(); i++){
            sum+=nums[i];
            max_i = max(sum,max_i);
            if (sum<0){
                sum = 0;
            }
        }
        return max_i;
    }
};
